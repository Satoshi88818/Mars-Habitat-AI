""" codes.py — Surface code definition with stabilizer permutation tracking. P0-2 Fix: dynamic_retile now records stabilizer_permutation and applies the inverse permutation to _syndrome_history and _noise_hist_buf so that temporal models consuming those buffers see a consistent stabilizer ordering after every retiling event. """ from __future__ import annotations import logging from dataclasses import dataclass, field from typing import List, Optional, Tuple import numpy as np logger = logging.getLogger(__name__) @dataclass class SurfaceCode: """ Rotated surface code of distance `d`. Attributes ---------- d : int Code distance (odd, >= 3). n_data : int Number of data qubits = d². n_stabilizers : int Number of stabilizers = d² - 1. stabilizer_permutation : np.ndarray Maps old stabilizer index → new stabilizer index after the last retile. Identity if no retile has occurred. _syndrome_history : np.ndarray Shape (history_len, n_stabilizers). Always in *current* stabilizer ordering. _noise_hist_buf : np.ndarray Shape (history_len, n_data). """ d: int = 5 history_len: int = 32 def __post_init__(self) -> None: if self.d < 3 or self.d % 2 == 0: raise ValueError("Code distance must be an odd integer >= 3.") self.n_data: int = self.d ** 2 self.n_stabilizers: int = self.d ** 2 - 1 self.stabilizer_permutation: np.ndarray = np.arange( self.n_stabilizers, dtype=np.int64 ) self._syndrome_history: np.ndarray = np.zeros( (self.history_len, self.n_stabilizers), dtype=np.int8 ) self._noise_hist_buf: np.ndarray = np.zeros( (self.history_len, self.n_data), dtype=np.float32 ) self._retile_count: int = 0 self._build_check_matrix() # ── Stabilizer matrix ─────────────────────────────────────────────────── def _build_check_matrix(self) -> None: """Build X- and Z-check matrices H_X, H_Z ∈ {0,1}^{n_stabilizers/2 × n_data}.""" d = self.d half = self.n_stabilizers // 2 self.H_X: np.ndarray = np.zeros((half, self.n_data), dtype=np.int8) self.H_Z: np.ndarray = np.zeros((half, self.n_data), dtype=np.int8) # Simplified nearest-neighbour check assignment (replace with full # rotated-code geometry for production use) for i in range(half): row, col = divmod(i, d - 1) for dq in [row * d + col, row * d + col + 1, (row + 1) * d + col, (row + 1) * d + col + 1]: if 0 <= dq < self.n_data: self.H_X[i, dq] = 1 self.H_Z[i, dq] = 1 # ── Syndrome recording ────────────────────────────────────────────────── def record_syndrome(self, syndrome: np.ndarray, noise: Optional[np.ndarray] = None) -> None: """ Shift syndrome history and append new measurement. """ assert syndrome.shape == (self.n_stabilizers,), ( f"Expected syndrome of shape ({self.n_stabilizers},), got {syndrome.shape}" ) self._syndrome_history = np.roll(self._syndrome_history, shift=1, axis=0) self._syndrome_history[0] = syndrome if noise is not None: self._noise_hist_buf = np.roll(self._noise_hist_buf, shift=1, axis=0) self._noise_hist_buf[0] = noise # ── Dynamic retiling (P0-2 fix) ───────────────────────────────────────── def dynamic_retile(self, new_d: Optional[int] = None) -> None: """ Perform a retiling operation (possibly changing code distance). After retiling the stabilizer *labels* change. This method: 1. Computes a random permutation of stabilizer indices (stands in for the real geometry-derived permutation in a production system). 2. Stores it as ``self.stabilizer_permutation``. 3. Applies the *inverse* permutation to ``_syndrome_history`` and ``_noise_hist_buf`` so that every temporal model sees consistent stabilizer ordering. Parameters ---------- new_d : int, optional New code distance after retiling. Unchanged if None. """ old_n = self.n_stabilizers if new_d is not None and new_d != self.d: if new_d < 3 or new_d % 2 == 0: raise ValueError("New distance must be an odd integer >= 3.") self.d = new_d self.n_data = new_d ** 2 self.n_stabilizers = new_d ** 2 - 1 self._build_check_matrix() # Resize history buffers to new stabilizer count new_hist = np.zeros( (self.history_len, self.n_stabilizers), dtype=np.int8 ) overlap = min(old_n, self.n_stabilizers) new_hist[:, :overlap] = self._syndrome_history[:, :overlap] self._syndrome_history = new_hist new_noise = np.zeros( (self.history_len, self.n_data), dtype=np.float32 ) overlap_d = min(old_n + 1, self.n_data) new_noise[:, :overlap_d] = self._noise_hist_buf[:, :overlap_d] self._noise_hist_buf = new_noise # Compute a permutation of current stabilizer indices perm = np.random.permutation(self.n_stabilizers).astype(np.int64) self.stabilizer_permutation = perm # Apply inverse permutation to history buffers inv_perm = np.argsort(perm) self._syndrome_history = self._syndrome_history[:, inv_perm] # noise buf is over data qubits, not stabilizers — no permutation needed self._retile_count += 1 logger.debug( "Retile #%d: d=%d, n_stab=%d, perm[:5]=%s", self._retile_count, self.d, self.n_stabilizers, perm[:5], ) # ── Convenience accessors ─────────────────────────────────────────────── @property def syndrome_history(self) -> np.ndarray: """Return syndrome history in current stabilizer ordering.""" return self._syndrome_history.copy() @property def noise_history(self) -> np.ndarray: return self._noise_hist_buf.copy()

""" physics.py — Battery physics (DoD, efficiency) and thermal / gas physics. """ from __future__ import annotations from dataclasses import dataclass, field from config import BatteryConfig @dataclass class Battery: config: BatteryConfig energy_kwh: float = 500.0 def charge(self, power_kw: float, dt: float = 1.0) -> float: """Charge battery; returns actual power absorbed (kW).""" headroom = self.config.capacity_kwh * self.config.dod_max - self.energy_kwh energy_in = min(power_kw * dt * self.config.charge_eff, max(0.0, headroom)) self.energy_kwh += energy_in return energy_in / dt def discharge(self, power_kw: float, dt: float = 1.0) -> float: """Discharge battery; returns actual power drawn (kW).""" energy_out = min( power_kw * dt / self.config.discharge_eff, self.energy_kwh ) self.energy_kwh -= energy_out return energy_out / dt def soc(self) -> float: """State of charge ∈ [0, 1].""" return self.energy_kwh / self.config.capacity_kwh @dataclass class Physics: thermal_mass: float = 5e6 # J/°C insulation_r: float = 5.0 # m²·K/W volume: float = 500.0 # m³ pressure: float = 101.3 # kPa external_temp: float = -60.0 # °C def heat_loss(self, internal_temp: float) -> float: """Conductive heat loss through habitat shell (W).""" return (internal_temp - self.external_temp) / self.insulation_r def update_temp( self, current: float, heater_power: float, dt: float = 1.0 ) -> float: """Euler step for habitat temperature.""" q_net = heater_power - self.heat_loss(current) return current + q_net * dt / self.thermal_mass def gas_leak( self, internal: float, target: float, leak_rate: float, structure: float ) -> float: """Partial pressure change due to structural leakage.""" return -leak_rate * (internal - target) * structure

""" decoders.py — Full decoder suite. P1: Belief Propagation + Neural BP (fully differentiable, often beats MWPM) P1: Cascade-CNN decoder P1: Syndrome Router (small policy → pick decoder per round) P0-3: E2E gradient from noise estimator through decoder (syndrome matching loss) Dependencies: pip install pymatching (for classical MWPM baseline) """ from __future__ import annotations import logging from typing import Dict, Optional, Tuple import torch import torch.nn as nn import torch.nn.functional as F from torch import Tensor logger = logging.getLogger(__name__) # ── Try to import MWPM ────────────────────────────────────────────────────── try: import pymatching HAS_PYMATCHING = True except ImportError: HAS_PYMATCHING = False logger.warning("pymatching not found. Classical MWPM will not be available.") # ── Classical MWPM decoder ────────────────────────────────────────────────── class MWPMDecoder: """Thin wrapper around pymatching for baseline MWPM decoding.""" def __init__(self, H: "np.ndarray") -> None: # type: ignore[name-defined] if not HAS_PYMATCHING: raise ImportError("Install pymatching for MWPM decoding.") import pymatching self.matcher = pymatching.Matching(H) def decode(self, syndrome: "np.ndarray") -> "np.ndarray": # type: ignore[name-defined] return self.matcher.decode(syndrome) # ── Neural Belief Propagation ─────────────────────────────────────────────── class NeuralBP(nn.Module): """ Learned Belief Propagation decoder (P1). Implements flooding BP with per-iteration learnable channel weights. Fully differentiable — gradients flow back to the noise estimator. Reference: Nachmani et al., "Hyper-graph-network decoders for block codes" Parameters ---------- n_checks : int n_data : int bp_iterations : int Number of unrolled BP iterations. """ def __init__( self, n_checks: int, n_data: int, bp_iterations: int = 8, ) -> None: super().__init__() self.n_checks = n_checks self.n_data = n_data self.bp_iterations = bp_iterations # Per-iteration per-edge weights (learnable) self.edge_weights = nn.Parameter( torch.ones(bp_iterations, n_checks, n_data) ) def forward( self, syndrome: Tensor, # (B, n_checks) ∈ {0, 1} llr_init: Tensor, # (B, n_data) log-likelihood ratios H: Tensor, # (n_checks, n_data) incidence matrix ) -> Tensor: # (B, n_data) updated LLRs """Run `bp_iterations` rounds of weighted flooding BP.""" llr = llr_init for t in range(self.bp_iterations): w = torch.sigmoid(self.edge_weights[t]) # (n_checks, n_data) # Variable → check messages v2c = llr.unsqueeze(1).expand(-1, self.n_checks, -1) * H.unsqueeze(0) # Check → variable messages (tanh rule) tanh_v2c = torch.tanh(v2c / 2.0 + 1e-8) # Exclude self (approx.: use full product; refine if needed) c2v_tanh = tanh_v2c.prod(dim=-1, keepdim=True).clamp(-1 + 1e-6, 1 - 1e-6) c2v = 2.0 * torch.atanh(c2v_tanh) * H.unsqueeze(0) # Weighted aggregation back to variable nodes llr = llr_init + (w.unsqueeze(0) * c2v).sum(dim=1) return llr # ── Cascade-CNN decoder ───────────────────────────────────────────────────── class CascadeCNN(nn.Module): """ Fast Cascade-CNN decoder for low-overhead, hardware-friendly decoding. Input : syndrome snapshot (B, 1, d, d-1) after reshaping. Output : logical correction probabilities (B, 2) [X-logical, Z-logical] """ def __init__(self, d: int) -> None: super().__init__() self.d = d self.net = nn.Sequential( nn.Conv2d(1, 32, kernel_size=3, padding=1), nn.ReLU(), nn.Conv2d(32, 64, kernel_size=3, padding=1), nn.ReLU(), nn.AdaptiveAvgPool2d(1), nn.Flatten(), nn.Linear(64, 32), nn.ReLU(), nn.Linear(32, 2), ) def forward(self, syndrome: Tensor) -> Tensor: """syndrome: (B, n_checks) — reshaped internally.""" B = syndrome.shape[0] half = syndrome.shape[1] // 2 # Reshape to 2D grid (crude; replace with actual check geometry) s = syndrome[:, :half].view(B, 1, self.d, self.d - 1).float() return self.net(s) # ── Syndrome Router (P1) ──────────────────────────────────────────────────── class SyndromeRouter(nn.Module): """ Small learned policy that routes each syndrome to the best decoder (P1). Outputs a soft decision (for training) or argmax (for inference): 0 → Neural-MWPM / Neural-BP 1 → Cascade-CNN 2 → Classical MWPM """ def __init__(self, n_checks: int) -> None: super().__init__() self.net = nn.Sequential( nn.Linear(n_checks, 64), nn.ReLU(), nn.Linear(64, 3), ) def forward(self, syndrome: Tensor) -> Tensor: """Returns logits (B, 3).""" return self.net(syndrome.float()) def route(self, syndrome: Tensor) -> int: """Hard routing decision for a single syndrome.""" with torch.no_grad(): return int(self.forward(syndrome.unsqueeze(0)).argmax(dim=-1).item()) # ── Syndrome matching loss (P0-3 E2E decoder gradient) ───────────────────── def syndrome_matching_loss( predicted_correction: Tensor, # (B, n_data) soft correction target_syndrome: Tensor, # (B, n_checks) observed syndrome H: Tensor, # (n_checks, n_data) ) -> Tensor: """ Compute BCE loss between predicted residual syndrome and zero syndrome. Enables gradient flow from decoder back into the noise estimator (P0-3). """ # Predicted residual syndrome after applying soft correction pred_syndrome = torch.sigmoid(H.unsqueeze(0) @ predicted_correction.unsqueeze(-1)) pred_syndrome = pred_syndrome.squeeze(-1) # (B, n_checks) target = target_syndrome.float() return F.binary_cross_entropy(pred_syndrome, target) # ── Composite decoder ──────────────────────────────────────────────────────── class CompositeDecoder(nn.Module): """ Wraps Neural-BP, Cascade-CNN, and optionally classical MWPM. Uses SyndromeRouter to dispatch (P1). P0-3: when cfg.e2e_grad_decoder=True the syndrome_matching_loss is computed and returned so the caller can add it to the total loss. """ def __init__( self, n_checks: int, n_data: int, d: int, bp_iterations: int = 8, H: Optional[Tensor] = None, e2e_grad_decoder: bool = True, e2e_decoder_loss_coeff: float = 0.05, ) -> None: super().__init__() self.e2e = e2e_grad_decoder self.coeff = e2e_decoder_loss_coeff self.neural_bp = NeuralBP(n_checks, n_data, bp_iterations) self.cascade_cnn = CascadeCNN(d) self.router = SyndromeRouter(n_checks) self.register_buffer("H", H if H is not None else torch.zeros(n_checks, n_data)) def forward( self, syndrome: Tensor, llr: Tensor, ) -> Tuple[Tensor, Tensor]: """ Returns ------- correction : Tensor (B, n_data) aux_loss : Tensor scalar (0.0 if e2e disabled) """ route_logits = self.router(syndrome) # (B, 3) # Soft-weighted combination during training for differentiability weights = torch.softmax(route_logits, dim=-1) # (B, 3) bp_out = self.neural_bp(syndrome, llr, self.H) # (B, n_data) cnn_out = self.cascade_cnn(syndrome) # (B, 2) → expand cnn_out_full = cnn_out[:, :1].expand_as(bp_out) # broadcast correction = ( weights[:, 0:1] * bp_out + weights[:, 1:2] * cnn_out_full # slot 2: classical MWPM (non-differentiable; weight goes to gradient estimator) ) aux_loss = torch.tensor(0.0) if self.e2e and self.training: aux_loss = self.coeff * syndrome_matching_loss(correction, syndrome, self.H) return correction, aux_loss

""" agents.py — DreamerV3-style world-model RL agent. P1 upgrade: replaces DQN with a compact world-model agent for better sample efficiency on long-horizon decoding.
